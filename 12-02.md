### Задание 1

CREATE USER 'sys_temp'@'localhost' IDENTIFIED BY '1234'; -- Создаем пользователя sys_temp

SELECT User FROM mysql.user; -- Выводим список пользователей

![Список пользователей](https://github.com/payfoxxx/sys-pattern-homework-12-01-hw/blob/main/img/12-02/Users_list.png)

GRANT ALL PRIVILEGES ON *.* TO 'sys_temp'@'localhost'; -- Даем все права пользователю sys_temp

SHOW GRANTS FOR 'sys_temp'@'localhost'; -- Запрос на вывод всех прав пользователя sys_temp

![Права пользователя](https://github.com/payfoxxx/sys-pattern-homework-12-01-hw/blob/main/img/12-02/grants_for_user.png)

ALTER USER 'sys_temp'@'localhost' IDENTIFIED WITH mysql_native_password BY '1234'; -- Меняем схему аутентификации

![Подключение к БД sys_temp](https://github.com/payfoxxx/sys-pattern-homework-12-01-hw/blob/main/img/12-02/connect_by_sys_temp.png)

### Задание 3

SET PERSIST partial_revokes = ON; -- Включаем частичный Revoke, чтобы можно было забрать выборочно права у тех, у кого *.\*

REVOKE INSERT,UPDATE,DELETE ON sakila.* FROM 'sys_temp'@'localhost'; -- Убираем права на внесение, обновление, удаление у пользователя sys_temp на базе данных sakila

SHOW GRANTS FOR 'sys_temp'@'localhost'; -- Выводим права пользователя sys_temp
